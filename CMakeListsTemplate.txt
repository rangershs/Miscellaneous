CMAKE_MINIMUM_REQUIRED(VERSION 2.8)
PROJECT(Demos)

set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -Wall -Wextra -Werror -Wconversion -Wno-unused-parameter")
set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -Wold-style-cast -Woverloaded-virtual -std=c++11")
SET(CMAKE_CXX_FLAGS_DEBUG "$ENV{CXXFLAGS} -o0 -g -ggdb")
SET(CMAKE_CXX_FLAGS_RELEASE "$ENV{CXXFLAGS} -o3")

AUX_SOURCE_DIRECTORY(. DIR_SRCS)
ADD_EXECUTABLE(demo.elf ${DIR_SRCS})
